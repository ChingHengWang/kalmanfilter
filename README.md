		mat HX_(1,1);

		mat P_TERM(2,2);
		mat R(1,1);
		mat I(2,2);

* Kalman core

		void KalmanFilter(){
		    X_=F*X;
		    X_.print("X_=F*X: ");
		    P_=F*P*F.t()+Q;
		    P_.print("P_=F*P*F'+Q: ");
		    K=(P_*H.t())/as_scalar(H*P_*H.t()+R);
		    K.print("K: ");
		    X=X_+K*(observe_Z-as_scalar(H*X_));
		    X.print("X: ");
		    estimate_Z=X(0,0);
		  }
* Initialize parameter
